\RequirePackage{csquotes}

\newcounter{thesis@airefs}[chapter]
\setcounter{thesis@airefs}{0}

\crefname{thesis@airefs}{AI quote}{AI quotes}

\def\thesis@aione{}
\newbool{thesis@aionetog}
\def\thesis@aitwo{}
\newbool{thesis@aitwotog}

\def\@aientry#1#2#3#4{
    \begin{tabularx}{\linewidth}{@{}lX@{}}
        \textbf{\translate{ai-tool}:}&#1\\
        \textbf{\translate{ai-place}:}&#2\\
        \textbf{\translate{ai-history}:}&#3\\
        \textbf{\translate{ai-sentiment}:}&#4
    \end{tabularx}
}

\def\aitypeone#1#2#3#4{
    \listadd\thesis@aione{{#1}{#2}{#3}{#4}}
    \global\booltrue{thesis@aionetog}
}

\def\aitypetwo#1#2#3#4{
    \listadd\thesis@aitwo{{#1}{#2}{#3}{#4}}
    \global\booltrue{thesis@aitwotog}
}



\def\printaistatement{
    \begingroup
    \translate{ai-description}
    \par\@sigskip\@signarea\thesis@voidpar\endgroup\bigskip\thesis@voidpar

    \section*{Instance Listing}
    \subsection*{Type 1 AI Systems}
    \ifbool{thesis@aionetog}{%
        \renewcommand{\do}[1]{\@aientry##1}
        \dolistloop{\thesis@aione}%
    }{%
        No use of type 1 AI tools reported.
    }


    \subsection*{Type 2 AI Systems}
    \ifbool{thesis@aitwotog}{%
        \renewcommand{\do}[1]{\@aientry##1}
        \dolistloop{\thesis@aitwo}%
    }{%
        No use of type 2 AI tools reported.
    }

}

\newcommand{\thesis@pagetarget}[2]{%
  \phantomsection%
  \label[thesis@airefs]{#1}%
  \hypertarget{#1}{#2}%
}

\newcommand\thesis@getcurrentref[1]{%
 \ifnumequal{\value{#1}}{0}
  {??}
  {\the\value{#1}}%
}

\def\thesis@aihelper#1#2#3#4#5{%4main 5back
    \def\arga{#1}%
    \def\argb{#2}%
    \def\argc{#3}%
    \edef\argd{#4}%
    \edef\arge{#5}%
    \expanded{%
        \noexpand\aitypetwo
        {\arga\noexpand\thesis@pagetarget{\arge}{}}%
        {\noexpand\hypersetup{hidelinks}\noexpand\hyperref[\argd]{AI quote~\expandafter\thesis@getcurrentref{chapter}.\expandafter\arabic{thesis@airefs}}\noexpand\sidenote{\noexpand\hyperref[\argd]{\textcolor{black}{\noexpand\hyperlinksymbol{} AI quote~\expandafter\thesis@getcurrentref{chapter}.\expandafter\arabic{thesis@airefs} on page~\noexpand\pageref*{\argd}}}}}%
        {\noexpand\argb}%
        {\noexpand\argc}%
    }
}

\def\aiquote#1#2#3#4{%
    \stepcounter{thesis@airefs}%
    \edef\thesis@mainjump{internal:main:\thethesis@airefs:\thechapter}%
    \edef\thesis@backjump{internal:back:\thethesis@airefs:\thechapter}%
    \enquote{\thesis@pagetarget{\thesis@mainjump}{#4}}\footnote{\hyperref[\thesis@backjump]{AI quote~\expandafter\thesis@getcurrentref{chapter}.\expandafter\arabic{thesis@airefs}, see page~\pageref*{\thesis@backjump}}}%
    \thesis@aihelper{#1}{#2}{#3}{\thesis@mainjump}{\thesis@backjump}%
}
