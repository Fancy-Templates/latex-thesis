\newglossaryentry{formal:external-factor}{%
  name={external factor}, plural={external factors},
  description={%
    A factor defined outside of the \gls{formal:model} and its expressions. Its change can yield completely different values inferred by the solver.
  },%
  shortdesc={Factors outside of a \gls{formal:model}[models] control}%
}

\newglossaryentry{def:behavioral-equivalence}{%
  name={behavioral equivalence},
  description={%
    Two \glspl{formal:model}~\(A\) and~\(B\) have the same observable behavior for all \textit{variables} they do have in common\ldots
  },%
  shortdesc={%
    All \textit{variables} in common evaluate to the same value for all external factors.%
  }%
}
